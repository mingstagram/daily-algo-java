public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        int M = sc.nextInt();

            } else if (sum == M) {
                count++;
                lt++;
                rt--;
            } else {
                rt--;
            }
        }

        System.out.println(count);
    }
